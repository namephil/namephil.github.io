斐波那契数列—–动态规划法

var fib = function (n) {
    let dp = new Array(n + 1).fill(0);
    dp[1] = 1;
    for (let i = 2; i <= n; i++) {
      dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
  };

合并二维有序数组成一维有序数组

var mergeTwoLists = function(l1, l2) {
    if (l1 === null) {
        return l2;
    } else if (l2 === null) {
        return l1;
    } else if (l1.val < l2.val) {
        l1.next = mergeTwoLists(l1.next, l2);
        return l1;
    } else {
        l2.next = mergeTwoLists(l1, l2.next);
        return l2;
    }
};

反转链表

var reverseList = function(head) {
    let prev = null;
    let curr = head;
    while (curr) {
        const next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }
    return prev;
};

判断链表是否有环

var hasCycle = function(head) {
    if(head ==null || head.next == null || head.next.next == null){return false}
    let n1 = head
    let nn = head.next
    while(n1 != nn){
        if(nn.next == null || nn.next.next == null){
            return false
        }
        n1 = n1.next
        nn = nn.next.next
    }
    return true
};
